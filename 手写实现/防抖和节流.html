    防抖和节流

    # 防抖和节流

    # 防抖

    /**
 * 防抖函数 - 延迟执行，在等待时间内多次调用仅执行最后一次
 * @param {Function} fn - 要执行的函数
 * @param {number} wait - 等待时间（毫秒）
 * @param {boolean} immediate - 是否立即执行（默认false）
 * @returns {Function} - 返回包装后的函数
 */
function debounce(fn, wait = 300, immediate = false) {
  let timer = null;
  let lastArgs = null;
  let lastThis = null;

  // 执行函数的内部方法
  function invokeFunc() {
    if (lastArgs) {
      fn.apply(lastThis, lastArgs);
      lastArgs = lastThis = null;
    }
  }

  return function (...args) {
    lastArgs = args;
    lastThis = this;

    // 如果设置了立即执行且尚未执行
    if (immediate && !timer) {
      invokeFunc();
    }

    // 清除之前的定时器，重新计时
    if (timer) {
      clearTimeout(timer);
      timer = null;
    }

    // 设置新的定时器
    timer = setTimeout(() => {
      timer = null;
      if (!immediate) {
        invokeFunc();
      }
    }, wait);
  };
}

// 示例用法
setInterval(() => {
  debounce(() => {
    console.log('执行函数');
  }, 1000)();
}, 10);

    # 节流

    /**
 * 节流函数 - 限制函数在一定时间内只能执行一次
 * @param {Function} fn - 要执行的函数
 * @param {number} wait - 执行间隔（毫秒）
 * @param {Object} options - 配置选项
 * @param {boolean} options.leading - 是否在开始时立即执行（默认true）
 * @param {boolean} options.trailing - 是否在结束后执行（默认true）
 * @returns {Function} - 返回包装后的函数
 */
function throttle(fn, wait = 300, options = {}) {
  const { leading = true, trailing = true } = options;
  let timer = null;
  let lastExecTime = 0;
  let lastArgs = null;
  let lastThis = null;

  // 执行函数
  function execute() {
    if (timer) {
      clearTimeout(timer);
      timer = null;
    }
    lastExecTime = Date.now();
    fn.apply(lastThis, lastArgs);
    lastArgs = lastThis = null;
  }

  // 设置延迟执行
  function scheduleTrailingCall() {
    timer = setTimeout(() => {
      if (trailing && lastArgs) {
        execute();
      }
      timer = null;
    }, wait - (Date.now() - lastExecTime));
  }

  return function (...args) {
    const context = this;
    const now = Date.now();
    const timeSinceLastExec = now - lastExecTime;

    // 保存当前参数和上下文
    lastArgs = args;
    lastThis = context;

    // 如果还没到下次执行时间
    if (timeSinceLastExec < wait) {
      // 如果还没有定时器，且允许 trailing 调用，则设置定时器
      if (!timer && trailing) {
        scheduleTrailingCall();
      }
      return;
    }

    // 如果超过执行时间
    if (leading) {
      // 如果允许 leading 调用，则立即执行
      execute();
    } else if (!timer && trailing) {
      // 否则设置定时器在等待时间结束后执行
      scheduleTrailingCall();
    } else {
      // 更新最后执行时间
      lastExecTime = now;
    }
  };
}

// 示例用法
const throttledScroll = throttle(() => console.log('节流滚动'), 500);

    # 总结

    主要区别总结

    特性 防抖 (Debounce) 节流 (Throttle)
    执行时机 仅执行最后一次调用 固定间隔执行一次
    适用场景 搜索框联想、窗口大小调整 滚动加载、按钮防重复点击
    核心机制 每次调用重置定时器 控制执行间隔
    配置选项 immediate（是否立即执行） leading/trailing（首尾执行控制）
